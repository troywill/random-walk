    def seconds_since_midnight(time)
      return time.hour * 3600 + time.min * 60 + time.sec
    end

    def steps_per_second
      hours = 14

      return ( 20000.0 / ( 14.0 * 3600.0 ) )
    end

    def steps(time)

      step_rate = steps_per_second
